== curl
:raw_base_init_lua_url: https://raw.githubusercontent.com/chubbyhippo/init.lua/refs/heads/main

[source,shell,subs="attributes"]
----
mkdir -p $HOME/.config/nvim && curl {raw_base_init_lua_url}/init.lua -o $HOME/.config/nvim/init.lua
----

== win

[source,powershell,subs="attributes"]
----
if(!(Test-Path -Path "$env:USERPROFILE\AppData\Local\nvim\")){New-Item -ItemType Directory -Force -Path "$env:USERPROFILE\AppData\Local\nvim\"}; Invoke-WebRequest -Uri "{raw_base_init_lua_url}/init.lua" -OutFile "$env:USERPROFILE\AppData\Local\nvim\init.lua"
----
